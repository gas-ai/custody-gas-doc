---
title: Order List
api: GET /api/mpc/tron/gas/record/list
description: "Retrieve specific order records through ID"
---

#### Request Parameters

<ParamField query="ids" type="string" required>
  Comma-separated `request_id` strings
</ParamField>

#### Response Parameters

<Expandable title="object">
  <ResponseField name="trade_no" type="string">
    Resource purchase transaction No
  </ResponseField>
  <ResponseField name="id" type="string">
    Unique transaction identifier
  </ResponseField>
  <ResponseField name="status" type="string">
    Status: 0 for order creation success, 1 for resource delegation success, 2
    for resource delegation failure 10 for reclaim success
  </ResponseField>
  <ResponseField name="receive_address" type="string">
    resource receive address
  </ResponseField>
  <ResponseField name="amount" type="string">
    amount
  </ResponseField>
  <ResponseField name="energy_num" type="string">
    energy num
  </ResponseField>
  <ResponseField name="delegate_energy_num" type="string">
    delegate energy num
  </ResponseField>
  <ResponseField name="reclaim_energy_num" type="string">
    reclaim energy num
  </ResponseField>
  <ResponseField name="item_list" type="string">
    子订单
    <Expandable title="object">
      <ResponseField name="energy_num" type="string">
        energy num
      </ResponseField>
      <ResponseField name="energy_price" type="string">
        energy price,unit SUN
      </ResponseField>
      <ResponseField name="energy_txid" type="string">
        energy txid
      </ResponseField>
      <ResponseField name="delegate_time" type="string">
        delegate time
      </ResponseField>
      <ResponseField name="resource_address" type="long">
        resource address
      </ResponseField>
      <ResponseField name="reclaim_txid" type="string">
        reclaim txid
      </ResponseField>
      <ResponseField name="reclaim_time" type="long">
        reclaim time
      </ResponseField>
    </Expandable>
  </ResponseField>
</Expandable>
