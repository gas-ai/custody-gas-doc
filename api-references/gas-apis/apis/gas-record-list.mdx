---
title: 查询记录
api: GET /api/mpc/tron/gas/record/list
description: "通过id获取特定的代理记录"
---

#### 请求参数数据结构
<ParamField query="ids" type="string" required>
    多个request_id的字符串，英文逗号分割，例:123,345
</ParamField>

#### 响应参数数据结构
<Expandable title="object">
    <ResponseField name="trade_no" type="string">订单ID</ResponseField>
    <ResponseField name="id" type="string">业务方ID</ResponseField>
    <ResponseField name="status" type="string">状态:0创建订单成功1代理资源成功2代理资源失败10回收成功</ResponseField>
    <ResponseField name="receive_address" type="string">资源接收地址</ResponseField>
    <ResponseField name="amount" type="string">支付金额</ResponseField>
    <ResponseField name="energy_num" type="string">能量数量</ResponseField>
    <ResponseField name="delegate_energy_num" type="string">已经代理数量</ResponseField>
    <ResponseField name="reclaim_energy_num" type="string">已回收数量</ResponseField>
    <ResponseField name="item_list" type="string">
        子订单
        <Expandable title="object">
            <ResponseField name="energy_num" type="string">能量代理数量</ResponseField>
            <ResponseField name="energy_price" type="string">能量价格,单位sun</ResponseField>
            <ResponseField name="energy_txid" type="string">能量代理TXID</ResponseField>
            <ResponseField name="delegate_time" type="string">代理时间</ResponseField>
            <ResponseField name="resource_address" type="long">发送资源地址</ResponseField>
            <ResponseField name="reclaim_txid" type="string">能量回收TXID</ResponseField>
            <ResponseField name="reclaim_time" type="long">能量回收时间</ResponseField>
        </Expandable>
    </ResponseField>
</Expandable>
