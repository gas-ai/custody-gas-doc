#### 请求参数数据结构

<ParamField query="receive_address" type="string" required>
    资源接收地址
</ParamField>
<ParamField query="address_to" type="string" required>
    转账到账地址，用于预估矿工费。
</ParamField>
<ParamField query="contract_address" type="string" required>
    合约地址，用于预估矿工费。
</ParamField>
<ParamField query="service_charge_type" type="string" required>
    租赁周期code :请求order/price接口返回分钟 小时 天 不同时间的费用code。
    <p>示例：10分钟:10010, 1小时:20001, 1天:30001</p>
</ParamField>


#### 响应参数数据结构

<Expandable title="object">
    <ResponseField name="contract_address" type="string">合约地址</ResponseField>
    <ResponseField name="address_to" type="string">转账到账地址</ResponseField>
    <ResponseField name="receive_address" type="string">资源接收地址</ResponseField>
    <ResponseField name="amount" type="decimal">订单金额</ResponseField>
    <ResponseField name="energy_amount" type="decimal">能量费用</ResponseField>
    <ResponseField name="active_amount" type="decimal">激活费用</ResponseField>
    <ResponseField name="energy_num" type="integer">能量数</ResponseField>
    <ResponseField name="energy_price" type="decimal">能量单价</ResponseField>
    <ResponseField name="service_charge_type" type="string">服务费code</ResponseField>
</Expandable>
