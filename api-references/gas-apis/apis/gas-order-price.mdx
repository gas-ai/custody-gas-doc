---
title: 获取下单价格
api: GET /api/mpc/tron/gas/order/price
description: "获取下单价格"
---
#### 请求参数数据结构
<ParamField query="resource_type" type="string" required>
  资源类型,默认为"energy"。例:"energy"或"net"
</ParamField>
<ParamField query="service_charge_type" type="string">
  租赁周期:不填返回所有支持的租赁信息:分钟:10010-10分钟,小时:20001-1小时,天:30001-1天
</ParamField>

#### 响应参数数据结构

<Expandable title="object">
    <ResponseField name="resource_type" type="string">资源类型</ResponseField>
    <ResponseField name="min_number" type="integer">最小下单数量</ResponseField>
    <ResponseField name="max_number" type="integer">最大下单数量</ResponseField>
    <ResponseField name="price_builder_list" type="list">
        价格列表
        <Expandable title="object">
            <ResponseField name="expire_min" type="integer">过期分钟数</ResponseField>
            <ResponseField name="service_charge_type" type="string">租赁周期code</ResponseField>
            <ResponseField name="price" type="decimal">价格</ResponseField>
            <ResponseField name="remaining_number" type="long">剩余可租赁数量</ResponseField>
        </Expandable>
    </ResponseField>

</Expandable>
